#crie um programa que armazene 5 váriaveis
# A, B, C, D, E 
A = int(input("Digite um número"))
B = int(input("Digite outro número "))
C = int(input("Digite outro número "))
D = int(input("Digite outro número "))
E = int(input("Digite outro número "))
# Calcular a área do triângulo
area_triangulo = (B * C) / 2
print("A área do triângulo é:", area_triangulo)
# perimetro do retangulo
perimetro_retangulo = 2 * (A + B + C + D)
print("o perimetro do retangulo é", perimetro_retangulo)
#circulo
pi = 3.14159
area_circulo = pi * E ** 2
print("A área do círculo é:", area_circulo)
